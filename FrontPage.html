        <!-- Input form for JSON data -->

        <form id="jsonForm">

            <textarea id="jsonData" placeholder="Enter JSON array..."></textarea>

            <br>

            <label for="filters">Select Filter:</label>

            <select id="filters" multiple>

                <option value="numbers">Numbers</option>

                <option value="alphabets">Alphabets</option>

                <option value="highest_lowercase_alphabet">Highest Lowercase Alphabet</option>

            </select>

            <br>

            <button type="submit">Submit</button>

        </form>


        <!-- Display response -->

        <div id="response">

            <h3>Response:</h3>

            <div id="responseContent"></div>

        </div>

    </div>


    <script>

        document.getElementById('jsonForm').addEventListener('submit', async function (event) {

            event.preventDefault();


            const jsonData = document.getElementById('jsonData').value;

            const filters = [...document.getElementById('filters').selectedOptions].map(option => option.value);


            // Prepare form data

            const formData = new FormData();

            formData.append('data', jsonData);


            try {

                const response = await fetch('/bfhl', {

                    method: 'POST',

                    body: formData

                });

                const result = await response.json();


                // Show filtered result

                let displayData = '';

                filters.forEach(filter => {

                    displayData += <p><strong>${filter}:</strong> ${JSON.stringify(result[filter] || 'No data')}</p>;

                });


                document.getElementById('responseContent').innerHTML = displayData;

            } catch (error) {

                console.error('Error:', error);

            }

        });

    </script>
